/**
 * 使用分析表对输入字符串进行语法分析（详细版本）
 * @param input_string 输入的字符串，符号之间用空格分隔
 * @param table 分析表（可以是LR(0)或LR(1)分析表）
 * @param grammar 文法定义
 * @param verbose 是否打印详细分析过程
 * @return true表示分析成功，false表示分析失败
 */
bool parse_string(const std::string& input_string, const LR0Table& table, const Grammar& grammar, bool verbose) {
    // 解析输入字符串为符号向量
    std::vector<std::string> input_symbols;  // 存储输入符号的向量
    std::stringstream ss(input_string);      // 创建字符串流
    std::string symbol;                      // 临时存储符号
    while (ss >> symbol) {                  // 逐个读取符号
        input_symbols.push_back(symbol);    // 添加到符号向量
    }
    
    // 初始化分析栈，压入初始状态0
    std::vector<StackElem> analysis_stack;  // 分析栈
    analysis_stack.emplace_back(0, "");     // 初始状态为0，符号为空
    
    // 输入符号索引
    size_t input_index = 0;                 // 当前处理的输入符号索引
    
    if (verbose) {  // 如果需要详细输出
        std::cout << "开始语法分析：" << std::endl;
        std::cout << "步骤\t栈\t\t\t输入\t\t\t动作" << std::endl;
        std::cout << "--------------------------------------------------------" << std::endl;
    }
    
    int step = 0;  // 步数计数器
    while (true) {
        step++;  // 增加步数
        
        if (verbose) {  // 如果需要详细输出
            // 打印当前状态
            std::cout << step << "\t";  // 打印步骤数
            for (const auto& elem : analysis_stack) {  // 遍历分析栈
                if (elem.symbol.empty()) {  // 如果符号为空
                    std::cout << "[" << elem.state << "] ";  // 只打印状态
                } else {  // 如果符号不为空
                    std::cout << "[" << elem.symbol << " " << elem.state << "] ";  // 打印符号和状态
                }
            }
            std::cout << "\t\t";  // 打印分隔符

            for (size_t i = input_index; i < input_symbols.size(); i++) {  // 遍历剩余输入符号
                std::cout << input_symbols[i] << " ";  // 打印输入符号
            }
            std::cout << "\t\t";  // 打印分隔符
        }

        // 获取栈顶状态
        int current_state = analysis_stack.back().state;  // 获取栈顶状态
        std::string current_input = input_symbols[input_index];  // 获取当前输入符号

        // 查找分析表
        auto action_it = table.action_table.find(std::to_string(current_state));  // 在action表中查找当前状态
        if (action_it == table.action_table.end()) {  // 如果状态不存在
            if (verbose) {
                std::cout << "错误：状态" << current_state << "不存在于分析表中" << std::endl;
            }
            return false;  // 返回失败
        }

        auto symbol_it = action_it->second.find(current_input);  // 在当前状态行中查找输入符号
        if (symbol_it == action_it->second.end()) {  // 如果输入符号不存在
            if (verbose) {
                std::cout << "错误：在状态" << current_state << "下，输入符号" << current_input << "无对应动作" << std::endl;
            }
            return false;  // 返回失败
        }

        TableValue action = symbol_it->second;  // 获取动作值

        switch (action.movement) {  // 根据动作类型执行相应操作
            case SHIFT: {  // 移进动作
                if (verbose) {
                    std::cout << "移进(s" << action.nextState << ")" << std::endl;
                }

                // 将当前输入符号和下一状态压入栈
                analysis_stack.emplace_back(action.nextState, current_input);

                input_index++;  // 移动输入指针
                break;
            }

            case REDUCE: {  // 规约动作
                // 获取产生式编号（这里假设nextState表示产生式编号）
                int production_index = action.nextState;
                if (production_index < 0 || production_index >= grammar.production.size()) {  // 检查产生式编号有效性
                    if (verbose) {
                        std::cout << "错误：产生式编号" << production_index << "无效" << std::endl;
                    }
                    return false;
                }

                const Production& production = grammar.production[production_index];  // 获取对应产生式
                if (verbose) {
                    std::cout << "规约(r" << production_index << "): " << production.left << " -> ";
                    for (size_t i = 0; i < production.right.size(); i++) {  // 遍历产生式右部
                        if (i > 0) std::cout << " ";  // 添加空格分隔符
                        std::cout << production.right[i];  // 打印产生式右部符号
                    }
                    std::cout << std::endl;
                }

                // 从栈中弹出|β|对元素（每对包含符号和状态）
                size_t beta_size = production.right.size();  // 获取产生式右部长度
                for (size_t i = 0; i < beta_size; i++) {  // 循环弹出对应数量的元素
                    if (!analysis_stack.empty()) {  // 检查栈是否为空
                        analysis_stack.pop_back();  // 弹出栈顶元素
                    } else {
                        if (verbose) {
                            std::cout << "错误：分析栈为空，无法执行规约操作" << std::endl;
                        }
                        return false;
                    }
                }

                // 获取当前栈顶状态
                if (analysis_stack.empty()) {  // 检查栈是否为空
                    if (verbose) {
                        std::cout << "错误：分析栈为空，无法获取栈顶状态" << std::endl;
                    }
                    return false;
                }
                int top_state = analysis_stack.back().state;  // 获取栈顶状态

                // 查找GOTO表
                auto goto_it = table.goto_table.find(std::to_string(top_state));  // 在GOTO表中查找栈顶状态
                if (goto_it == table.goto_table.end()) {  // 如果状态不存在
                    if (verbose) {
                        std::cout << "错误：状态" << top_state << "不存在于GOTO表中" << std::endl;
                    }
                    return false;
                }

                auto goto_symbol_it = goto_it->second.find(production.left);  // 查找产生式左部符号
                if (goto_symbol_it == goto_it->second.end()) {  // 如果符号不存在
                    if (verbose) {
                        std::cout << "错误：在状态" << top_state << "下，非终结符" << production.left << "无对应GOTO状态" << std::endl;
                    }
                    return false;
                }

                std::string next_state_str = goto_symbol_it->second;  // 获取下一状态字符串
                if (next_state_str == "-1") {  // 检查状态是否有效
                    if (verbose) {
                        std::cout << "错误：在状态" << top_state << "下，非终结符" << production.left << "的GOTO状态为无效值" << std::endl;
                    }
                    return false;
                }

                int next_state = std::stoi(next_state_str);  // 转换为整数

                // 将非终结符和下一状态压入栈
                analysis_stack.emplace_back(next_state, production.left);

                break;
            }

            case ACCEPT: {  // 接受动作
                if (verbose) {
                    std::cout << "接受" << std::endl;
                    std::cout << "语法分析成功！" << std::endl;
                }
                return true;  // 返回成功
            }

            case ERR: {  // 错误动作
                if (verbose) {
                    std::cout << "错误：在状态" << current_state << "下，输入符号" << current_input << "对应错误动作" << std::endl;
                }
                return false;  // 返回失败
            }

            default: {  // 未知动作类型
                if (verbose) {
                    std::cout << "未知动作类型" << std::endl;
                }
                return false;  // 返回失败
            }
        }

        // 检查是否栈大小过大（防止无限循环）
        if (analysis_stack.size() > 1000) {  // 如果栈大小超过限制
            if (verbose) {
                std::cout << "错误：分析栈过大，可能存在无限循环" << std::endl;
            }
            return false;
        }
    }
}

#ifndef TASK9_LR0_ANALYSIS_H  // 防止重复包含的头文件保护

#include <unordered_map>      // 包含无序映射库
#include <vector>             // 包含向量库

struct LR1Item;  // 前向声明LR1项目结构

// 动作枚举：移进、规约、接受、错误
enum Movement {
    SHIFT, REDUCE, ACCEPT, ERR
};

// 分析栈的元素结构
struct StackElem {
    int state = -1;          // 状态编号，默认为-1
    std::string symbol;      // 符号

    StackElem() = default;   // 默认构造函数
    StackElem(const int state, std::string symbol) : state(state), symbol(std::move(symbol)) {}  // 参数构造函数
};

// 分析表内的值结构
struct TableValue {
    Movement movement = ERR;  // 动作类型，默认为错误
    int nextState = -1;       // 下一状态，默认为-1

    TableValue() = default;   // 默认构造函数
    TableValue(const Movement movement, const int nextState) : movement(movement), nextState(nextState) {}  // 参数构造函数
};

// LR(0)分析表结构
// 用二维map存，这样可以直接通过LR0Table[状态][(非)终结符]访问对应的值
struct LR0Table {
    std::unordered_map<std::string, std::unordered_map<std::string, TableValue>> action_table;  // action表
    std::unordered_map<std::string, std::unordered_map<std::string, std::string>> goto_table;   // goto表

    // 默认空表，不提供任何带初值的构造函数
    LR0Table() = default;
};

// 产生式结构
struct Production {
    std::string left;                    // 左部符号
    std::vector<std::string> right;      // 右部符号向量

    Production() = default;              // 默认构造函数
    Production(std::string left, std::string right) {  // 从单个右部符号构造
        this->left = std::move(left);    // 移动左部符号
        this->right.push_back(std::move(right));  // 添加右部符号
    }
    Production(std::string left, std::vector<std::string> right) : left(std::move(left)), right(std::move(right)) {}  // 从右部符号向量构造

    // 重载比较运算符，用于查找特定产生式
    bool operator==(const Production& other) const {
        return left == other.left && right == other.right;  // 比较左右部是否相等
    }
};

// 文法结构
struct Grammar {
    std::vector<std::string> terminal_sym;    // 终结符向量
    std::vector<std::string> non_terminal_sym; // 非终结符向量
    std::vector<Production> production;        // 产生式向量
    std::string start_sym;                    // 起始符号
};

// DFA状态结构
struct DFAState {
    std::string state_name;                    // 状态名称
    std::vector<Production> project_set;       // 项目集

    DFAState(std::string name, const std::vector<Production>& projects)  // 构造函数
        : state_name(std::move(name)), project_set(projects) {}
};

// 状态转换函数结构
struct StateTransFunc {
    DFAState current_state;                    // 当前状态
    std::string input_sym;                     // 输入符号
    std::vector<DFAState> next_state;          // 下一状态向量

    StateTransFunc() = delete;                 // 删除默认构造函数
    StateTransFunc(DFAState current_state, std::string input_sym, std::vector<DFAState> next_state) :  // 参数构造函数
            current_state(std::move(current_state)), input_sym(std::move(input_sym)), next_state(std::move(next_state)) {}
};

// LR1状态结构
struct LR1State {
    std::string state_name;                    // 状态名称
    std::vector<LR1Item> items;                // LR1项目向量

    LR1State(std::string name, const std::vector<LR1Item>& its)  // 构造函数
        : state_name(std::move(name)), items(its) {}
};

// 添加一个新的结构体用于LR1的状态转换
struct LR1StateTransFunc {
    LR1State current_state;                    // 当前LR1状态
    std::string input_sym;                     // 输入符号
    std::vector<LR1State> next_state;          // 下一LR1状态向量

    LR1StateTransFunc() = default;             // 默认构造函数
    LR1StateTransFunc(LR1State current_state, std::string input_sym, std::vector<LR1State> next_state) :  // 参数构造函数
            current_state(std::move(current_state)), input_sym(std::move(input_sym)), next_state(std::move(next_state)) {}
};

// DFA结构
struct DFA {
    std::vector<DFAState> states;              // 状态向量
    std::vector<std::string> alphabet;         // 字母表
    std::vector<StateTransFunc> transitions;   // 转换函数向量
    std::vector<DFAState> start_state;         // 初态向量
    std::vector<DFAState> final_state;         // 终态向量
};

// LR1项目结构
struct LR1Item {
    Production production;                     // 产生式
    std::string lookahead;                     // 向前看符号
    size_t dot_pos;                          // 点号位置

    LR1Item(const Production& p, const std::string& la, size_t pos)  // 构造函数
        : production(p), lookahead(la), dot_pos(pos) {}

    bool operator==(const LR1Item& other) const {  // 重载比较运算符
        return production.left == other.production.left &&  // 比较产生式左部
               production.right == other.production.right &&  // 比较产生式右部
               lookahead == other.lookahead &&  // 比较向前看符号
               dot_pos == other.dot_pos;  // 比较点号位置
    }
};

// 用于LR1项目的哈希函数
struct LR1ItemHash {
    std::size_t operator()(const LR1Item& item) const {  // 哈希函数
        std::size_t h1 = std::hash<std::string>{}(item.production.left);  // 哈希左部
        std::size_t h2 = std::hash<std::string>{}(item.lookahead);        // 哈希向前看符号
        std::size_t h3 = std::hash<size_t>{}(item.dot_pos);              // 哈希点号位置

        for (const auto& sym : item.production.right) {  // 遍历右部符号
            h1 ^= std::hash<std::string>{}(sym) + 0x9e3779b9 + (h1 << 6) + (h1 >> 2);  // 计算复合哈希
        }

        return h1 ^ (h2 << 1) ^ (h3 << 2);  // 返回组合哈希值
    }
};

// LR1分析器结构
struct LR1DFA {
    std::vector<LR1State> states;              // LR1状态向量
    std::vector<std::string> alphabet;         // 字母表
    std::vector<LR1StateTransFunc> transitions; // LR1状态转换向量
    std::vector<LR1State> start_state;         // 初态向量
    std::vector<LR1State> final_state;         // 终态向量
};

#include "LR_Analysis.h"  // 包含头文件

#include <fstream>        // 包含文件流库

void input_data(Grammar& grammar) {  // 从文件读取文法数据
    try {
        std::ifstream input_file("../input.txt");  // 打开输入文件
        std::string line;                          // 临时存储行数据
        std::string temp;                          // 临时存储符号
        // 这里用的vector，可以自动扩展长度，不用手动定义数组长度，输入的数字就不必要了。
        // 输入非终结符
        std::getline(input_file, line);            // 先把数字读出来丢了
        std::getline(input_file, line);            // 再读取非终结符
        std::istringstream iss(line);              // 创建字符串流
        while(iss >> temp) {                      // 逐个读取符号
            grammar.non_terminal_sym.push_back(temp);  // 添加到非终结符向量
        }
        std::getline(input_file, line);            // 跳过空行

        // 输入终结符
        std::getline(input_file, line);            // 先把数字读出来丢了
        std::getline(input_file, line);            // 再读取终结符
        iss = std::istringstream(line);            // 重新初始化字符串流
        while(iss >> temp) {                      // 逐个读取符号
            grammar.terminal_sym.push_back(temp);  // 添加到终结符向量
        }
        std::getline(input_file, line);            // 跳过空行

        // 输入产生式
        std::getline(input_file, line);            // 读出数字
        int num_production = std::stoi(line);      // 转换为整数
        for (int i = 0; i < num_production; i++) {  // 循环读取产生式
            std::getline(input_file, line);        // 读取一行
            iss = std::istringstream(line);        // 重新初始化字符串流

            std::vector<std::string> syms;         // 存储符号向量
            while (iss >> temp) {                 // 逐个读取符号
                syms.push_back(temp);             // 添加到符号向量
            }

            // 添加一点错误检查
            if (syms.size() >= 3 && syms[1] == "->") {  // 检查产生式格式
                // 跳过第一个元素(左侧非终结符)和"->"，从第三个元素开始读取产生式右部
                Production new_p;                  // 创建新产生式
                new_p.left = syms[0];              // 设置左部
                for (int count = 2; count < syms.size(); count++) {  // 遍历右部符号
                    new_p.right.push_back(syms[count]);  // 添加到右部
                }
                grammar.production.push_back(new_p);  // 添加到产生式向量
            } else {
                std::cerr << "产生式格式错误:！原始行数据如下：" << line << std::endl;  // 错误提示
            }
        }
        std::getline(input_file, line);            // 跳过空行

        // 输入起始符号
        std::getline(input_file, line);            // 读取起始符号
        grammar.start_sym = line;                  // 设置起始符号

    } catch (const std::exception& e) {            // 捕获异常
        std::cerr << "文件读取失败：" << e.what() << std::endl;  // 错误提示

        // 如果文件读取失败，使用默认文法进行测试
        std::cout << "使用默认文法进行测试：" << std::endl;
        grammar.non_terminal_sym = {"S", "A"};     // 设置默认非终结符
        grammar.terminal_sym = {"a", "b", "#"};    // 设置默认终结符
        grammar.start_sym = "S";                   // 设置默认起始符号

        // S -> A
        Production p1;                             // 创建产生式1
        p1.left = "S";                             // 设置左部
        p1.right = {"A"};                          // 设置右部
        grammar.production.push_back(p1);          // 添加到产生式向量

        // A -> aA
        Production p2;                             // 创建产生式2
        p2.left = "A";                             // 设置左部
        p2.right = {"a", "A"};                     // 设置右部
        grammar.production.push_back(p2);          // 添加到产生式向量

        // A -> b
        Production p3;                             // 创建产生式3
        p3.left = "A";                             // 设置左部
        p3.right = {"b"};                          // 设置右部
        grammar.production.push_back(p3);          // 添加到产生式向量
    }
}


void generate_project(  // 生成项目
    std::unordered_map<std::string, std::vector<Production>>& project_map,  // 项目映射
    Grammar& grammar)  // 文法
{
    // 添加拓广产生式到开头
    bool flag = false;  // 标志是否已存在拓广产生式
    for (const auto& production : grammar.production) {  // 遍历所有产生式
        if (production.left == "S'") {  // 检查是否为拓广产生式
            flag = true;  // 设置标志
        }
    }
    if (!flag) {  // 如果不存在拓广产生式
        Production start_production;  // 创建拓广产生式
        start_production.left = "S'";  // 设置左部为S'
        start_production.right.push_back(grammar.start_sym);  // 右部为原起始符号
        grammar.production.insert(grammar.production.begin(), start_production);  // 插入到开头
    }

    // 构造项目
    for (const auto& production : grammar.production) {  // 遍历所有产生式
        // 为每个产生式生成所有可能的LR(0)项目
        for (size_t i = 0; i <= production.right.size(); ++i) {  // 遍历所有可能的点号位置
            Production new_production;  // 创建新项目
            new_production.left = production.left;  // 设置左部
            new_production.right.clear();  // 清空右部

            // 在位置i插入点号
            for (size_t j = 0; j < production.right.size(); ++j) {  // 遍历原产生式右部
                if (j == i) {  // 如果到达点号位置
                    new_production.right.emplace_back(".");  // 插入点号
                }
                new_production.right.push_back(production.right[j]);  // 添加原符号
            }
            // 如果点号在末尾
            if (i == production.right.size()) {  // 如果点号应该在末尾
                new_production.right.emplace_back(".");  // 在末尾添加点号
            }

            project_map[new_production.left].push_back(new_production);  // 添加到项目映射
        }
    }
}


void generate_project_set(  // 生成项目集
    DFA& dfa,  // DFA
    const std::unordered_map<std::string, std::vector<Production>>& project_map,  // 项目映射
    const Grammar& grammar)  // 文法
{
    // 初始化项目集族，从初始项目集开始
    std::vector<std::vector<Production>> project_sets;  // 存储所有项目集
    std::vector<bool> processed_sets;  // 标记已处理的项目集

    // 创建初始项目集 I0 = Closure({S' -> .S})
    std::vector<Production> initial_project;  // 初始项目集
    // 添加起始项目
    initial_project.push_back(project_map.at("S'")[0]);  // 添加S' -> .S项目

    // 计算初始项目集的闭包
    std::vector<Production> closure_initial = calc_closure(initial_project, project_map, grammar);  // 计算闭包
    project_sets.push_back(closure_initial);  // 添加到项目集向量
    processed_sets.push_back(false);  // 初始状态未处理

    // 收集所有符号（终结符和非终结符）
    std::set<std::string> all_symbols;  // 所有符号集合
    for (const auto& term : grammar.terminal_sym) {  // 遍历终结符
        all_symbols.insert(term);  // 添加到集合
    }
    for (const auto& non_term : grammar.non_terminal_sym) {  // 遍历非终结符
        all_symbols.insert(non_term);  // 添加到集合
    }

    // 存储状态转移映射
    std::unordered_map<size_t, std::unordered_map<std::string, size_t>> transitions_map;  // 状态转移映射

    // 对每个项目集和每个符号计算GO函数
    std::queue<size_t> work_queue;    // 待处理的项目集索引队列
    work_queue.push(0);  // 将初始状态加入队列

    while (!work_queue.empty()) {  // 当队列不为空时
        size_t current_index = work_queue.front();  // 获取队首元素
        work_queue.pop();  // 弹出队首元素

        if (processed_sets[current_index]) {  // 如果已处理过
            continue;  // 跳过
        }
        processed_sets[current_index] = true;  // 标记为已处理

        // 对每个符号计算GO函数
        for (const auto& symbol : all_symbols) {  // 遍历所有符号
            std::vector<Production> go_result = GO(project_sets[current_index], symbol, project_map, grammar);  // 计算GO函数

            // 如果GO结果为空，跳过
            if (go_result.empty()) {  // 如果结果为空
                continue;  // 跳过
            }

            // 检查是否是新项目集
            bool is_new = true;  // 标记是否为新项目集
            size_t existing_index = 0;  // 存储已存在项目集的索引
            for (size_t i = 0; i < project_sets.size(); ++i) {  // 遍历已有项目集
                if (is_project_set_equal(go_result, project_sets[i])) {  // 比较是否相等
                    is_new = false;  // 不是新项目集
                    existing_index = i;  // 记录索引
                    break;  // 跳出循环
                }
            }

            if (is_new) {  // 如果是新项目集
                project_sets.push_back(go_result);  // 添加到项目集向量
                processed_sets.push_back(false);  // 新状态未处理
                work_queue.push(project_sets.size() - 1);  // 将新状态加入队列
                existing_index = project_sets.size() - 1;  // 更新索引
            }

            // 记录状态转移
            transitions_map[current_index][symbol] = existing_index;  // 记录转移关系
        }
    }

    // 构建DFA
    // 添加状态
    for (size_t i = 0; i < project_sets.size(); ++i) {  // 遍历所有项目集
        dfa.states.emplace_back(std::to_string(i), project_sets[i]);  // 创建DFA状态
    }

    // 添加字母表
    dfa.alphabet.insert(dfa.alphabet.end(), all_symbols.begin(), all_symbols.end());  // 添加所有符号到字母表

    // 设置初态
    if (!dfa.states.empty()) {  // 如果状态不为空
        dfa.start_state.push_back(dfa.states[0]);  // 设置第一个状态为初态
    }

    // 设置终态（包含归约项目的状态）
    for (size_t i = 0; i < dfa.states.size(); ++i) {  // 遍历所有状态
        bool is_final = false;  // 终态标志
        for (const auto& project : dfa.states[i].project_set) {  // 遍历项目集
            // 检查点是否在产生式末尾
            if (!project.right.empty() && project.right.back() == ".") {  // 如果点在末尾
                is_final = true;  // 设置终态标志
                break;  // 跳出循环
            }
        }
        if (is_final) {  // 如果是终态
            dfa.final_state.push_back(dfa.states[i]);  // 添加到终态向量
        }
    }

    // 添加状态转移关系
    for (const auto& transition_entry : transitions_map) {  // 遍历转移映射
        size_t from_state = transition_entry.first;  // 起始状态
        for (const auto& symbol_transition : transition_entry.second) {  // 遍历符号转移
            std::string symbol = symbol_transition.first;  // 符号
            size_t to_state = symbol_transition.second;  // 目标状态

            if (from_state < dfa.states.size() && to_state < dfa.states.size()) {  // 检查索引有效性
                StateTransFunc trans_func(dfa.states[from_state], symbol, {dfa.states[to_state]});  // 创建转移函数
                dfa.transitions.push_back(trans_func);  // 添加到转移向量
            }
        }
    }
}

std::vector<Production> GO(const std::vector<Production>& project_set,  // GO函数
                          const std::string& symbol,  // 符号
                          const std::unordered_map<std::string, std::vector<Production>>& project_map,  // 项目映射
                          const Grammar& grammar) {  // 文法
    std::vector<Production> result;  // 结果向量

    // 找出项目集中所有点号后是symbol的项目，并将点号右移一位
    for (const auto& project : project_set) {  // 遍历项目集
        for (size_t i = 0; i < project.right.size(); ++i) {  // 遍历产生式右部
            if (project.right[i] == "." && i + 1 < project.right.size() && project.right[i + 1] == symbol) {  // 如果点号后是目标符号
                Production moved_project = project;  // 复制项目
                // 将点号右移
                moved_project.right[i] = symbol;  // 将符号移到点号位置
                moved_project.right[i + 1] = ".";  // 将点号移到下一个位置
                result.push_back(moved_project);  // 添加到结果
            }
        }
    }

    // 计算结果的闭包
    return calc_closure(result, project_map, grammar);  // 返回闭包
}


std::vector<Production> calc_closure(const std::vector<Production>& project_set,  // 计算闭包
                                   const std::unordered_map<std::string, std::vector<Production>>& project_map,  // 项目映射
                                   const Grammar& grammar)  // 文法
{
    std::vector<Production> result = project_set;  // 初始化结果为输入项目集
    bool changed = true;  // 变化标志

    while (changed) {  // 当有变化时继续循环
        changed = false;  // 重置变化标志
        std::vector<Production> new_projects;  // 新项目向量

        // 查找所有点号后是非终结符的项目
        for (const auto& project : result) {  // 遍历当前结果
            for (size_t i = 0; i < project.right.size(); ++i) {  // 遍历产生式右部
                if (project.right[i] == "." && i + 1 < project.right.size()) {  // 如果点号不在末尾
                    std::string next_symbol = project.right[i + 1];  // 获取点号后的符号

                    // 检查是否为非终结符
                    if (std::find(grammar.non_terminal_sym.begin(), grammar.non_terminal_sym.end(), next_symbol)  // 查找是否为非终结符
                        != grammar.non_terminal_sym.end()) {

                        // 添加该非终结符的所有产生式的项目（点号在最前面）
                        auto it = project_map.find(next_symbol);  // 在项目映射中查找
                        if (it != project_map.end()) {  // 如果找到
                            for (const auto& prod : it->second) {  // 遍历对应的所有产生式
                                // 检查这个产生式是否是点号在最前面
                                if (!prod.right.empty() && prod.right[0] == ".") {  // 如果点号在最前面
                                    // 检查是否已存在
                                    bool exists = false;  // 存在标志
                                    for (const auto& existing : result) {  // 遍历当前结果
                                        if (existing.left == prod.left && existing.right == prod.right) {  // 比较是否相同
                                            exists = true;  // 设置存在标志
                                            break;  // 跳出循环
                                        }
                                    }

                                    if (!exists) {  // 如果不存在
                                        new_projects.push_back(prod);  // 添加到新项目向量
                                        changed = true;  // 设置变化标志
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }

        // 添加新项目到结果中
        result.insert(result.end(), new_projects.begin(), new_projects.end());  // 将新项目合并到结果中
    }

    return result;  // 返回闭包结果
}

// 比较两个项目集是否相等
bool is_project_set_equal(const std::vector<Production>& set1, const std::vector<Production>& set2) {
    if (set1.size() != set2.size()) {  // 如果大小不同
        return false;  // 返回false
    }

    // 排序后再比较，确保顺序不影响结果
    std::vector<Production> sorted_set1 = set1;  // 复制集合1
    std::vector<Production> sorted_set2 = set2;  // 复制集合2

    // 自定义排序函数
    auto prod_cmp = [](const Production& a, const Production& b) {  // 比较函数
        if (a.left != b.left) return a.left < b.left;  // 比较左部
        if (a.right.size() != b.right.size()) return a.right.size() < b.right.size();  // 比较右部长度
        return a.right < b.right;  // 比较右部内容
    };

    std::sort(sorted_set1.begin(), sorted_set1.end(), prod_cmp);  // 排序集合1
    std::sort(sorted_set2.begin(), sorted_set2.end(), prod_cmp);  // 排序集合2

    for (size_t i = 0; i < sorted_set1.size(); ++i) {  // 遍历排序后的集合
        if (sorted_set1[i].left != sorted_set2[i].left ||  // 比较左部
            sorted_set1[i].right != sorted_set2[i].right) {  // 比较右部
            return false;  // 如果不相等返回false
        }
    }

    return true;  // 返回true
}

bool generate_lr0_table(const DFA& dfa, const Grammar& grammar, LR0Table& lr0_table) {  // 生成LR(0)分析表
    // 初始化分析表
    for (const auto& dfa_state : dfa.states) {  // 遍历DFA的所有状态
        std::string state = dfa_state.state_name;  // 获取状态名

        // 初始化action表
        for (const auto& terminal : grammar.terminal_sym) {  // 遍历所有终结符
            lr0_table.action_table[state][terminal] = TableValue(ERR, -1);  // 设置为错误
        }
        // 添加结束符#的处理
        lr0_table.action_table[state]["#"] = TableValue(ERR, -1);  // 设置结束符为错误

        // 初始化goto表
        for (const auto& non_terminal : grammar.non_terminal_sym) {  // 遍历所有非终结符
            lr0_table.goto_table[state][non_terminal] = "-1";  // 设置为无效
        }
    }

    // 遍历每个DFA状态来填充分析表
    for (size_t state_idx = 0; state_idx < dfa.states.size(); ++state_idx) {  // 遍历所有DFA状态
        std::string state = dfa.states[state_idx].state_name;  // 获取状态名
        const std::vector<Production>& project_set = dfa.states[state_idx].project_set;  // 获取项目集

        // 遍历当前状态的项目集来填充action表和goto表
        for (const auto& project : project_set) {  // 遍历项目集
            // 查找点号的位置
            size_t dot_pos = std::find(project.right.begin(), project.right.end(), ".") - project.right.begin();  // 查找点号位置

            if (dot_pos >= project.right.size()) continue;  // 如果没有点号，跳过

            // 情况1：点号不在最右边 - 移进项
            if (dot_pos + 1 < project.right.size()) {  // 如果点号不在末尾
                std::string next_symbol = project.right[dot_pos + 1];  // 获取点号后的符号

                // 如果是终结符，添加移进动作
                if (std::find(grammar.terminal_sym.begin(), grammar.terminal_sym.end(), next_symbol)  // 查找是否为终结符
                    != grammar.terminal_sym.end()) {

                    // 计算GO函数来确定下一状态
                    std::vector<Production> go_result = GO(project_set, next_symbol,  // 计算GO函数
                        std::unordered_map<std::string, std::vector<Production>>(), grammar);

                    // 查找GO结果对应的状态索引
                    int next_state_idx = -1;  // 下一状态索引
                    for (size_t i = 0; i < dfa.states.size(); ++i) {  // 遍历所有DFA状态
                        if (is_project_set_equal(go_result, dfa.states[i].project_set)) {  // 比较项目集是否相等
                            next_state_idx = i;  // 记录状态索引
                            break;  // 跳出循环
                        }
                    }

                    if (next_state_idx != -1) {  // 如果找到状态
                        lr0_table.action_table[state][next_symbol] = TableValue(SHIFT, next_state_idx);  // 设置移进动作
                    }
                }
                // 如果是非终结符，添加到goto表
                else if (std::find(grammar.non_terminal_sym.begin(), grammar.non_terminal_sym.end(), next_symbol)  // 查找是否为非终结符
                         != grammar.non_terminal_sym.end()) {

                    // 计算GO函数来确定下一状态
                    std::vector<Production> go_result = GO(project_set, next_symbol,  // 计算GO函数
                        std::unordered_map<std::string, std::vector<Production>>(), grammar);

                    // 查找GO结果对应的状态索引
                    int next_state_idx = -1;  // 下一状态索引
                    for (size_t i = 0; i < dfa.states.size(); ++i) {  // 遍历所有DFA状态
                        if (is_project_set_equal(go_result, dfa.states[i].project_set)) {  // 比较项目集是否相等
                            next_state_idx = i;  // 记录状态索引
                            break;  // 跳出循环
                        }
                    }

                    if (next_state_idx != -1) {  // 如果找到状态
                        lr0_table.goto_table[state][next_symbol] = std::to_string(next_state_idx);  // 设置GOTO表
                    }
                }
            }
            // 情况2：点号在最右边 - 规约项或接受项
            else {  // 点号在最右边
                // 如果是拓广产生式 S' -> S.，则为接受状态
                if (project.left == "S'" && project.right.size() == 2 &&  // 检查是否为拓广产生式
                    project.right[0] == "." && project.right[1] == grammar.start_sym) {
                    lr0_table.action_table[state]["#"] = TableValue(ACCEPT, -1);  // 设置接受动作
                } else {  // 规约动作
                    // 规约动作 - 对所有终结符（包括#）进行规约
                    std::vector<std::string> original_right;  // 原始右部
                    for (const auto& sym : project.right) {  // 遍历产生式右部
                        if (sym != ".") {  // 跳过点号
                            original_right.push_back(sym);  // 添加到原始右部
                        }
                    }

                    // 查找产生式在grammar.production中的索引
                    int production_index = -1;  // 产生式索引
                    for (size_t i = 0; i < grammar.production.size(); ++i) {  // 遍历所有产生式
                        if (grammar.production[i].left == project.left &&  // 比较左部
                            grammar.production[i].right == original_right) {  // 比较右部
                            production_index = i;  // 记录索引
                            break;  // 跳出循环
                        }
                    }

                    if (production_index != -1) {  // 如果找到产生式
                        // 对所有终结符（包括#）进行规约
                        for (const auto& terminal : grammar.terminal_sym) {  // 遍历所有终结符
                            if (lr0_table.action_table[state][terminal].movement == ERR) {  // 如果当前为错误
                                lr0_table.action_table[state][terminal] = TableValue(REDUCE, production_index);  // 设置规约动作
                            }
                        }
                        if (lr0_table.action_table[state]["#"].movement == ERR) {  // 对结束符进行规约
                            lr0_table.action_table[state]["#"] = TableValue(REDUCE, production_index);  // 设置规约动作
                        }
                    }
                }
            }
        }
    }
}

/**
 * 检查文法是否为LR(0)文法
 * @param dfa DFA状态机
 * @param grammar 文法定义
 * @return true表示是LR(0)文法，false表示不是
 */
bool is_lr0_grammar(const DFA& dfa, const Grammar& grammar) {
    for (const auto& dfa_state : dfa.states) {  // 遍历所有DFA状态
        const std::vector<Production>& project_set = dfa_state.project_set;  // 获取项目集

        // 分别统计移进项目和归约项目
        std::vector<Production> shift_projects;   // 移进项目
        std::vector<Production> reduce_projects;  // 归约项目

        for (const auto& project : project_set) {  // 遍历项目集
            // 查找点号的位置
            size_t dot_pos = std::find(project.right.begin(), project.right.end(), ".") - project.right.begin();  // 查找点号位置

            if (dot_pos >= project.right.size()) continue;  // 如果没有点号，跳过

            // 情况1：点号不在最右边 - 移进项目
            if (dot_pos + 1 < project.right.size()) {  // 如果点号不在末尾
                shift_projects.push_back(project);  // 添加到移进项目向量
            }
            // 情况2：点号在最右边 - 归约项目
            else {  // 点号在末尾
                reduce_projects.push_back(project);  // 添加到归约项目向量
            }
        }

        // 如果同时存在移进项目和归约项目，则存在移进-归约冲突
        if (!shift_projects.empty() && !reduce_projects.empty()) {  // 如果同时有移进和归约项目
            return false;  // 返回false
        }

        // 如果存在多个不同的归约项目，则存在归约-归约冲突
        if (reduce_projects.size() > 1) {  // 如果有多个归约项目
            // 检查是否为不同的产生式（有不同的左部或右部）
            for (size_t i = 0; i < reduce_projects.size(); i++) {  // 遍历归约项目
                for (size_t j = i + 1; j < reduce_projects.size(); j++) {  // 比较不同项目
                    // 比较两个归约项目的原始产生式（去除点号）
                    std::vector<std::string> right1, right2;  // 两个项目的右部
                    for (const auto& sym : reduce_projects[i].right) {  // 遍历第一个项目右部
                        if (sym != ".") {  // 跳过点号
                            right1.push_back(sym);  // 添加到右部1
                        }
                    }
                    for (const auto& sym : reduce_projects[j].right) {  // 遍历第二个项目右部
                        if (sym != ".") {  // 跳过点号
                            right2.push_back(sym);  // 添加到右部2
                        }
                    }

                    // 如果是不同的产生式，则存在归约-归约冲突
                    if (reduce_projects[i].left != reduce_projects[j].left || right1 != right2) {  // 比较产生式是否不同
                        return false;  // 返回false
                    }
                }
            }
        }
    }

    return true;  // 返回true
}

bool is_production_equals(const Production& p1, const Production& p2) {  // 比较产生式是否相等
    if (p1.left != p2.left) {  // 比较左部
        return false;  // 不相等返回false
    }
    if (p1.right.size() != p2.right.size()) {  // 比较右部长度
        return false;  // 长度不同返回false
    }
    for (size_t i = 0; i < p1.right.size(); i++) {  // 遍历右部
        if (p1.right[i] != p2.right[i]) {  // 比较每个符号
            return false;  // 不相等返回false
        }
    }
    return true;  // 相等返回true
}

// LR1 相关函数实现
std::vector<std::string> get_first(const std::vector<std::string>& symbols, const Grammar& grammar) {  // 获取FIRST集
    std::vector<std::string> result;  // 结果向量
    if (symbols.empty()) {  // 如果符号串为空
        result.push_back("#");  // 返回空串符号
        return result;  // 返回结果
    }

    // 处理符号串的FIRST集
    bool can_be_empty = true;  // 可空标志
    for (const auto& symbol : symbols) {  // 遍历符号串
        // 如果是终结符，直接加入
        if (std::find(grammar.terminal_sym.begin(), grammar.terminal_sym.end(), symbol) != grammar.terminal_sym.end()) {  // 检查是否为终结符
            result.push_back(symbol);  // 添加到结果
            can_be_empty = false;  // 不可空
            break;  // 跳出循环
        }
        // 如果是非终结符
        else if (std::find(grammar.non_terminal_sym.begin(), grammar.non_terminal_sym.end(), symbol) != grammar.non_terminal_sym.end()) {  // 检查是否为非终结符
            bool has_epsilon = false;  // 有空产生式标志
            // 找到所有以该非终结符为左部的产生式
            for (const auto& prod : grammar.production) {  // 遍历所有产生式
                if (prod.left == symbol) {  // 如果左部匹配
                    // 如果是空产生式
                    if (prod.right.empty() || (prod.right.size() == 1 && prod.right[0] == "#")) {  // 检查是否为空产生式
                        has_epsilon = true;  // 设置有空产生式标志
                    } else {  // 非空产生式
                        // 递归计算右部的FIRST集
                        auto first_of_right = get_first(prod.right, grammar);  // 递归计算FIRST集
                        for (const auto& f : first_of_right) {  // 遍历FIRST集
                            if (f != "#") {  // 如果不是空串符号
                                result.push_back(f);  // 添加到结果
                            } else {  // 如果是空串符号
                                has_epsilon = true;  // 设置有空产生式标志
                            }
                        }
                    }
                }
            }
            
            if (!has_epsilon) {  // 如果没有空产生式
                can_be_empty = false;  // 设置不可空
                break;  // 跳出循环
            }
        }
    }
    
    if (can_be_empty) {  // 如果可空
        result.push_back("#");  // 添加空串符号
    }

    // 去重
    std::sort(result.begin(), result.end());  // 排序
    result.erase(std::unique(result.begin(), result.end()), result.end());  // 去重

    return result;  // 返回结果
}

std::vector<LR1Item> lr1_closure(const std::vector<LR1Item>& items,  // LR1闭包
                                 const std::vector<Production>& productions,  // 产生式
                                 const Grammar& grammar) {  // 文法
    std::vector<LR1Item> result = items;  // 初始化结果
    bool changed = true;  // 变化标志

    while (changed) {  // 当有变化时继续
        changed = false;  // 重置变化标志

        for (size_t i = 0; i < result.size(); i++) {  // 遍历结果中的项目
            const LR1Item& item = result[i];  // 获取当前项目
            // 如果点号不在末尾且后面跟着非终结符
            if (item.dot_pos < item.production.right.size()) {  // 检查点号位置
                std::string next_symbol = item.production.right[item.dot_pos];  // 获取点号后的符号

                // 检查是否为非终结符
                if (std::find(grammar.non_terminal_sym.begin(), grammar.non_terminal_sym.end(), next_symbol) != grammar.non_terminal_sym.end()) {  // 检查是否为非终结符
                    // 计算剩余部分β的FIRST集
                    std::vector<std::string> beta_symbols;  // β符号向量
                    for (size_t j = item.dot_pos + 1; j < item.production.right.size(); ++j) {  // 遍历点号后的符号
                        beta_symbols.push_back(item.production.right[j]);  // 添加到β符号向量
                    }
                    
                    // 添加当前的向前看符号
                    beta_symbols.push_back(item.lookahead);  // 添加向前看符号
                    
                    // 计算FIRST(β a)
                    std::vector<std::string> first_set = get_first(beta_symbols, grammar);  // 计算FIRST集

                    // 为所有以next_symbol为左部的产生式创建LR1项目
                    for (const auto& prod : productions) {  // 遍历产生式
                        if (prod.left == next_symbol) {  // 如果左部匹配
                            // 为FIRST集中的每个符号创建一个项目
                            for (const auto& la : first_set) {  // 遍历FIRST集
                                LR1Item new_item(prod, la, 0);  // 创建新项目

                                // 检查是否已存在
                                bool exists = false;  // 存在标志
                                for (const auto& existing : result) {  // 遍历结果
                                    if (existing.production.left == new_item.production.left &&  // 比较产生式左部
                                        existing.production.right == new_item.production.right &&  // 比较产生式右部
                                        existing.lookahead == new_item.lookahead &&  // 比较向前看符号
                                        existing.dot_pos == new_item.dot_pos) {  // 比较点号位置
                                        exists = true;  // 设置存在标志
                                        break;  // 跳出循环
                                    }
                                }

                                if (!exists) {  // 如果不存在
                                    result.push_back(new_item);  // 添加到结果
                                    changed = true;  // 设置变化标志
                                }
                            }
                        }
                    }
                }
            }
        }
    }

    return result;  // 返回结果
}

std::vector<LR1Item> lr1_go(const std::vector<LR1Item>& items,  // LR1 GO函数
                            const std::string& symbol,  // 符号
                            const std::vector<Production>& productions,  // 产生式
                            const Grammar& grammar) {  // 文法
    std::vector<LR1Item> result;  // 结果向量

    // 找出所有点号后是symbol的项目，并将点号右移一位
    for (const auto& item : items) {  // 遍历项目
        if (item.dot_pos < item.production.right.size() &&  // 检查点号位置
            item.production.right[item.dot_pos] == symbol) {  // 检查点号后的符号
            result.push_back(LR1Item(item.production, item.lookahead, item.dot_pos + 1));  // 创建新项目，点号右移
        }
    }

    // 计算闭包
    return lr1_closure(result, productions, grammar);  // 返回闭包
}

bool is_lr1_item_set_equal(const std::vector<LR1Item>& set1, const std::vector<LR1Item>& set2) {  // 比较LR1项目集是否相等
    if (set1.size() != set2.size()) {  // 检查大小
        return false;  // 不同返回false
    }

    // 排序后再比较
    std::vector<LR1Item> sorted_set1 = set1;  // 复制集合1
    std::vector<LR1Item> sorted_set2 = set2;  // 复制集合2

    auto item_cmp = [](const LR1Item& a, const LR1Item& b) {  // 比较函数
        if (a.production.left != b.production.left) return a.production.left < b.production.left;  // 比较产生式左部
        if (a.production.right != b.production.right) return a.production.right < b.production.right;  // 比较产生式右部
        if (a.lookahead != b.lookahead) return a.lookahead < b.lookahead;  // 比较向前看符号
        return a.dot_pos < b.dot_pos;  // 比较点号位置
    };

    std::sort(sorted_set1.begin(), sorted_set1.end(), item_cmp);  // 排序集合1
    std::sort(sorted_set2.begin(), sorted_set2.end(), item_cmp);  // 排序集合2

    for (size_t i = 0; i < sorted_set1.size(); ++i) {  // 遍历排序后的集合
        if (sorted_set1[i].production.left != sorted_set2[i].production.left ||  // 比较产生式左部
            sorted_set1[i].production.right != sorted_set2[i].production.right ||  // 比较产生式右部
            sorted_set1[i].lookahead != sorted_set2[i].lookahead ||  // 比较向前看符号
            sorted_set1[i].dot_pos != sorted_set2[i].dot_pos) {  // 比较点号位置
            return false;  // 如果不相等返回false
        }
    }

    return true;  // 相等返回true
}

void generate_lr1_states(LR1DFA& lr1_dfa, const Grammar& grammar) {  // 生成LR1状态
    // 初始化项目集族
    std::vector<std::vector<LR1Item>> item_sets;  // 项目集向量
    std::vector<bool> processed_sets;  // 已处理标志向量

    // 创建初始项目集 I0 = Closure({[S' -> .S, #]})
    std::vector<LR1Item> initial_items;  // 初始项目向量
    for (const auto& prod : grammar.production) {  // 遍历产生式
        if (prod.left == "S'") {  // 找到拓广产生式
            initial_items.emplace_back(prod, "#", 0);  // 创建初始项目
            break;  // 跳出循环
        }
    }

    std::vector<LR1Item> closure_initial = lr1_closure(initial_items, grammar.production, grammar);  // 计算闭包
    item_sets.push_back(closure_initial);  // 添加到项目集向量
    processed_sets.push_back(false);  // 初始状态未处理

    // 收集所有符号
    std::set<std::string> all_symbols;  // 所有符号集合
    for (const auto& term : grammar.terminal_sym) {  // 遍历终结符
        all_symbols.insert(term);  // 添加到集合
    }
    for (const auto& non_term : grammar.non_terminal_sym) {  // 遍历非终结符
        all_symbols.insert(non_term);  // 添加到集合
    }

    // 存储状态转移映射
    std::unordered_map<size_t, std::unordered_map<std::string, size_t>> transitions_map;  // 状态转移映射

    // BFS 构建DFA
    std::queue<size_t> work_queue;  // 工作队列
    work_queue.push(0);  // 将初始状态加入队列

    while (!work_queue.empty()) {  // 当队列不为空
        size_t current_index = work_queue.front();  // 获取队首元素
        work_queue.pop();  // 弹出队首元素

        if (processed_sets[current_index]) {  // 如果已处理
            continue;  // 跳过
        }
        processed_sets[current_index] = true;  // 标记为已处理

        // 对每个符号计算GO函数
        for (const auto& symbol : all_symbols) {  // 遍历所有符号
            std::vector<LR1Item> go_result = lr1_go(item_sets[current_index], symbol, grammar.production, grammar);  // 计算GO函数

            if (go_result.empty()) {  // 如果结果为空
                continue;  // 跳过
            }

            // 检查是否是新项目集
            bool is_new = true;  // 新项目集标志
            size_t existing_index = 0;  // 已存在索引
            for (size_t i = 0; i < item_sets.size(); ++i) {  // 遍历项目集
                if (is_lr1_item_set_equal(go_result, item_sets[i])) {  // 比较是否相等
                    is_new = false;  // 不是新项目集
                    existing_index = i;  // 记录索引
                    break;  // 跳出循环
                }
            }

            if (is_new) {  // 如果是新项目集
                item_sets.push_back(go_result);  // 添加到项目集向量
                processed_sets.push_back(false);  // 新状态未处理
                work_queue.push(item_sets.size() - 1);  // 将新状态加入队列
                existing_index = item_sets.size() - 1;  // 更新索引
            }

            // 记录状态转移
            transitions_map[current_index][symbol] = existing_index;  // 记录转移关系
        }
    }

    // 构建LR1DFA
    for (size_t i = 0; i < item_sets.size(); ++i) {  // 遍历项目集
        lr1_dfa.states.emplace_back(std::to_string(i), item_sets[i]);  // 创建状态
    }

    lr1_dfa.alphabet.insert(lr1_dfa.alphabet.end(), all_symbols.begin(), all_symbols.end());  // 添加字母表

    if (!lr1_dfa.states.empty()) {  // 如果状态不为空
        lr1_dfa.start_state.push_back(lr1_dfa.states[0]);  // 设置初态
    }

    // 设置终态（包含归约项目的状态）
    for (size_t i = 0; i < lr1_dfa.states.size(); ++i) {  // 遍历所有状态
        bool is_final = false;  // 终态标志
        for (const auto& item : lr1_dfa.states[i].items) {  // 遍历项目
            if (item.dot_pos == item.production.right.size()) {  // 检查点号是否在末尾
                is_final = true;  // 设置终态标志
                break;  // 跳出循环
            }
        }
        if (is_final) {  // 如果是终态
            lr1_dfa.final_state.push_back(lr1_dfa.states[i]);  // 添加到终态向量
        }
    }

    // 添加状态转移关系
    for (const auto& transition_entry : transitions_map) {  // 遍历转移映射
        size_t from_state = transition_entry.first;  // 起始状态
        for (const auto& symbol_transition : transition_entry.second) {  // 遍历符号转移
            std::string symbol = symbol_transition.first;  // 符号
            size_t to_state = symbol_transition.second;  // 目标状态

            if (from_state < lr1_dfa.states.size() && to_state < lr1_dfa.states.size()) {  // 检查索引有效性
                std::vector<LR1State> next_states;  // 下一状态向量
                next_states.push_back(lr1_dfa.states[to_state]);  // 添加下一状态
                LR1StateTransFunc trans_func(lr1_dfa.states[from_state], symbol, next_states);  // 创建转移函数
                lr1_dfa.transitions.push_back(trans_func);  // 添加到转移向量
            }
        }
    }
}

void generate_lr1_analysis_table(const LR1DFA& lr1_dfa, const Grammar& grammar, LR0Table& lr1_table) {  // 生成LR1分析表
    // 初始化分析表
    for (const auto& lr1_state : lr1_dfa.states) {  // 遍历所有LR1状态
        std::string state = lr1_state.state_name;  // 获取状态名

        // 初始化action表
        for (const auto& terminal : grammar.terminal_sym) {  // 遍历所有终结符
            lr1_table.action_table[state][terminal] = TableValue(ERR, -1);  // 设置为错误
        }
        lr1_table.action_table[state]["#"] = TableValue(ERR, -1);  // 设置结束符为错误

        // 初始化goto表
        for (const auto& non_terminal : grammar.non_terminal_sym) {  // 遍历所有非终结符
            lr1_table.goto_table[state][non_terminal] = "-1";  // 设置为无效
        }
    }

    // 遍历每个状态来填充分析表
    for (size_t state_idx = 0; state_idx < lr1_dfa.states.size(); ++state_idx) {  // 遍历所有状态
        std::string state = lr1_dfa.states[state_idx].state_name;  // 获取状态名
        const std::vector<LR1Item>& items = lr1_dfa.states[state_idx].items;  // 获取项目向量

        for (const auto& item : items) {  // 遍历项目
            if (item.dot_pos < item.production.right.size()) {  // 如果点号不在末尾
                // 移进项目
                std::string next_symbol = item.production.right[item.dot_pos];  // 获取点号后的符号

                // 如果是终结符，添加移进动作
                if (std::find(grammar.terminal_sym.begin(), grammar.terminal_sym.end(), next_symbol) != grammar.terminal_sym.end()) {  // 检查是否为终结符
                    // 计算GO函数来确定下一状态
                    std::vector<LR1Item> go_result = lr1_go(items, next_symbol, grammar.production, grammar);  // 计算GO函数

                    // 查找GO结果对应的状态索引
                    int next_state_idx = -1;  // 下一状态索引
                    for (size_t i = 0; i < lr1_dfa.states.size(); ++i) {  // 遍历所有状态
                        if (is_lr1_item_set_equal(go_result, lr1_dfa.states[i].items)) {  // 比较项目集是否相等
                            next_state_idx = i;  // 记录索引
                            break;  // 跳出循环
                        }
                    }

                    if (next_state_idx != -1) {  // 如果找到状态
                        lr1_table.action_table[state][next_symbol] = TableValue(SHIFT, next_state_idx);  // 设置移进动作
                    }
                }
                // 如果是非终结符，添加到goto表
                else if (std::find(grammar.non_terminal_sym.begin(), grammar.non_terminal_sym.end(), next_symbol) != grammar.non_terminal_sym.end()) {  // 检查是否为非终结符
                    // 计算GO函数来确定下一状态
                    std::vector<LR1Item> go_result = lr1_go(items, next_symbol, grammar.production, grammar);  // 计算GO函数

                    // 查找GO结果对应的状态索引
                    int next_state_idx = -1;  // 下一状态索引
                    for (size_t i = 0; i < lr1_dfa.states.size(); ++i) {  // 遍历所有状态
                        if (is_lr1_item_set_equal(go_result, lr1_dfa.states[i].items)) {  // 比较项目集是否相等
                            next_state_idx = i;  // 记录索引
                            break;  // 跳出循环
                        }
                    }

                    if (next_state_idx != -1) {  // 如果找到状态
                        lr1_table.goto_table[state][next_symbol] = std::to_string(next_state_idx);  // 设置GOTO表
                    }
                }
            } else {  // 点号在末尾，规约项目或接受项目
                if (item.production.left == "S'") {  // 如果是拓广产生式
                    // 接受项目
                    lr1_table.action_table[state][item.lookahead] = TableValue(ACCEPT, -1);  // 设置接受动作
                } else {  // 规约项目
                    // 查找产生式在grammar.production中的索引
                    int production_index = -1;  // 产生式索引
                    for (size_t i = 0; i < grammar.production.size(); ++i) {  // 遍历所有产生式
                        if (grammar.production[i].left == item.production.left &&  // 比较左部
                            grammar.production[i].right == item.production.right) {  // 比较右部
                            production_index = i;  // 记录索引
                            break;  // 跳出循环
                        }
                    }

                    if (production_index != -1) {  // 如果找到产生式
                        // 只对当前向前看符号进行规约
                        if (lr1_table.action_table[state][item.lookahead].movement == ERR) {  // 如果当前为错误
                            lr1_table.action_table[state][item.lookahead] = TableValue(REDUCE, production_index);  // 设置规约动作
                        }
                    }
                }
            }
        }
    }
}

/**
 * 检查文法是否为LR(1)文法
 * @param lr1_dfa LR(1) DFA状态机
 * @param grammar 文法定义
 * @return true表示是LR(1)文法，false表示不是
 */
bool is_lr1_grammar(const LR1DFA& lr1_dfa, const Grammar& grammar) {
    for (const auto& lr1_state : lr1_dfa.states) {  // 遍历所有LR1状态
        const std::vector<LR1Item>& items = lr1_state.items;  // 获取项目向量

        // 使用map来记录每个符号对应的动作
        std::unordered_map<std::string, std::vector<TableValue>> symbol_actions;  // 符号动作映射

        // 分析每个项目
        for (const auto& item : items) {  // 遍历项目
            if (item.dot_pos < item.production.right.size()) {  // 如果点号不在末尾
                // 移进项目
                std::string next_symbol = item.production.right[item.dot_pos];  // 获取点号后的符号

                // 如果是终结符，添加移进动作
                if (std::find(grammar.terminal_sym.begin(), grammar.terminal_sym.end(), next_symbol) != grammar.terminal_sym.end()) {  // 检查是否为终结符
                    // 查找GO函数来确定下一状态
                    std::vector<LR1Item> go_result = lr1_go(items, next_symbol, grammar.production, grammar);  // 计算GO函数

                    // 查找GO结果对应的状态索引
                    int next_state_idx = -1;  // 下一状态索引
                    for (size_t i = 0; i < lr1_dfa.states.size(); ++i) {  // 遍历所有状态
                        if (is_lr1_item_set_equal(go_result, lr1_dfa.states[i].items)) {  // 比较项目集是否相等
                            next_state_idx = i;  // 记录索引
                            break;  // 跳出循环
                        }
                    }

                    if (next_state_idx != -1) {  // 如果找到状态
                        symbol_actions[item.lookahead].push_back(TableValue(SHIFT, next_state_idx));  // 添加移进动作
                    }
                }
            } else {  // 点号在末尾，归约项目或接受项目
                if (item.production.left == "S'") {  // 如果是拓广产生式
                    // 接受项目
                    symbol_actions[item.lookahead].push_back(TableValue(ACCEPT, -1));  // 添加接受动作
                } else {  // 规约项目
                    // 查找产生式在grammar.production中的索引
                    int production_index = -1;  // 产生式索引
                    for (size_t i = 0; i < grammar.production.size(); ++i) {  // 遍历所有产生式
                        if (grammar.production[i].left == item.production.left &&  // 比较左部
                            grammar.production[i].right == item.production.right) {  // 比较右部
                            production_index = i;  // 记录索引
                            break;  // 跳出循环
                        }
                    }

                    if (production_index != -1) {  // 如果找到产生式
                        symbol_actions[item.lookahead].push_back(TableValue(REDUCE, production_index));  // 添加规约动作
                    }
                }
            }
        }

        // 检查每个符号是否存在冲突
        for (const auto& symbol_action_pair : symbol_actions) {  // 遍历符号动作映射
            const std::vector<TableValue>& actions = symbol_action_pair.second;  // 获取动作向量

            if (actions.size() > 1) {  // 如果有多个动作
                // 存在多个动作，检查是否为冲突
                bool has_shift = false;  // 有移进标志
                bool has_reduce = false;  // 有规约标志
                int reduce_production = -1;  // 规约产生式

                for (const auto& action : actions) {  // 遍历动作
                    if (action.movement == SHIFT) {  // 如果是移进
                        has_shift = true;  // 设置移进标志
                    } else if (action.movement == REDUCE) {  // 如果是规约
                        if (!has_reduce) {  // 如果还没有规约
                            reduce_production = action.nextState;  // 记录产生式
                            has_reduce = true;  // 设置规约标志
                        } else if (reduce_production != action.nextState) {  // 如果规约产生式不同
                            // 归约-归约冲突
                            return false;  // 返回false
                        }
                    } else if (action.movement == ACCEPT) {  // 如果是接受
                        has_shift = true;  // ACCEPT视为移进动作
                    }
                }

                // 如果同时有移进和规约，则存在移进-规约冲突
                if (has_shift && has_reduce) {  // 如果同时有移进和规约
                    return false;  // 返回false
                }
            }
        }
    }

    return true;  // 返回true
}

void generate_lr1_table(const DFA& dfa, const Grammar& grammar, LR0Table& lr1_table) {  // 生成LR1表
    LR1DFA lr1_dfa;  // 创建LR1DFA

    if (is_lr1_grammar(lr1_dfa, grammar)) {  // 检查是否为LR(1)文法
        std::cout << "\n文法是 LR(1) 文法！" << std::endl;  // 输出提示
    } else {
        std::cout << "\n文法不是 LR(1) 文法！" << std::endl;  // 输出提示
    }

    generate_lr1_states(lr1_dfa, grammar);  // 生成LR1状态
    generate_lr1_analysis_table(lr1_dfa, grammar, lr1_table);  // 生成LR1分析表

    // 打印LR1分析表
    std::cout << "\nLR(1) 分析表:\n";  // 打印标题
    std::cout << "ACTION 表:\n";  // 打印ACTION表标题
    std::cout << "State\t";  // 打印状态列标题
    for (const auto& terminal : grammar.terminal_sym) {  // 遍历终结符
        std::cout << terminal << "\t";  // 打印终结符标题
    }
    std::cout << "#\t";  // 打印结束符标题
    std::cout << "\n";  // 换行

    for (const auto& lr1_state : lr1_dfa.states) {  // 遍历LR1状态
        std::string state = lr1_state.state_name;  // 获取状态名
        std::cout << state << "\t";  // 打印状态名
        for (const auto& terminal : grammar.terminal_sym) {  // 遍历终结符
            TableValue val = lr1_table.action_table.at(state).at(terminal);  // 获取表值
            switch (val.movement) {  // 根据动作类型输出
                case SHIFT:  // 移进
                    std::cout << "s" << val.nextState << "\t";  // 输出s+状态号
                    break;
                case REDUCE:  // 规约
                    std::cout << "r" << val.nextState << "\t";  // 输出r+产生式号
                    break;
                case ACCEPT:  // 接受
                    std::cout << "acc\t";  // 输出acc
                    break;
                case ERR:  // 错误
                    std::cout << "err\t";  // 输出err
                    break;
            }
        }

        // 处理#符号
        TableValue val = lr1_table.action_table.at(state).at("#");  // 获取结束符表值
        switch (val.movement) {  // 根据动作类型输出
            case SHIFT:  // 移进
                std::cout << "s" << val.nextState << "\t";  // 输出s+状态号
                break;
            case REDUCE:  // 规约
                std::cout << "r" << val.nextState << "\t";  // 输出r+产生式号
                break;
            case ACCEPT:  // 接受
                std::cout << "acc\t";  // 输出acc
                break;
            case ERR:  // 错误
                std::cout << "err\t";  // 输出err
                break;
        }
        std::cout << "\n";  // 换行
    }

    // 打印GOTO表
    std::cout << "\nGOTO 表:\n";  // 打印标题
    std::cout << "State\t";  // 打印状态列标题
    for (const auto& non_terminal : grammar.non_terminal_sym) {  // 遍历非终结符
        std::cout << non_terminal << "\t";  // 打印非终结符标题
    }
    std::cout << "\n";  // 换行

    for (const auto& lr1_state : lr1_dfa.states) {  // 遍历LR1状态
        std::string state = lr1_state.state_name;  // 获取状态名
        std::cout << state << "\t";  // 打印状态名
        for (const auto& non_terminal : grammar.non_terminal_sym) {  // 遍历非终结符
            std::string val = lr1_table.goto_table.at(state).at(non_terminal);  // 获取GOTO表值
            if (val != "-1") {  // 如果不是无效值
                std::cout << val << "\t";  // 输出状态号
            } else {  // 如果是无效值
                std::cout << "err\t";  // 输出err
            }
        }
        std::cout << "\n";  // 换行
    }
}
