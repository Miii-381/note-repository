// 定义分析动作的枚举类型
enum Movement {
    SHIFT,  // 移进动作
    REDUCE, // 规约动作
    ACCEPT, // 接受动作 (成功)
    ERR     // 错误动作
};

// 分析栈中的元素结构体
struct StackElem {
    int state = -1;         // 当前状态编号
    std::string symbol;     // 对应的符号 (终结符或非终结符)
    StackElem() = default;  // 默认构造函数
    // 带参数的构造函数
    StackElem(const int state, std::string symbol) : state(state), symbol(std::move(symbol)) {}
};

// 分析表中每个单元格的值的结构体
struct TableValue {
    Movement movement = ERR; // 动作类型 (移进、规约等)
    int nextState = -1;      // 下一状态 (移进时为状态号，规约时为产生式索引)
    TableValue() = default;   // 默认构造函数
    // 带参数的构造函数
    TableValue(const Movement movement, const int nextState) : movement(movement), nextState(nextState) {}
};

// LR(0)分析表的结构体
struct LR0Table {
    // Action 表: 使用二维无序映射，键为 [状态][符号]，值为 TableValue
    // 第一层 key 是状态名 (string)，第二层 key 是符号 (string)
    std::unordered_map<std::string, std::unordered_map<std::string, TableValue>> action_table;
    // Goto 表: 使用二维无序映射，键为 [状态][非终结符]，值为下一状态 (string)
    std::unordered_map<std::string, std::unordered_map<std::string, std::string>> goto_table;
    LR0Table() = default; // 默认构造函数
};

// 文法产生式的结构体
struct Production {
    std::string left;               // 产生式的左部 (非终结符)
    std::vector<std::string> right; // 产生式的右部 (符号序列)
    Production() = default;         // 默认构造函数
    // 构造函数：从一个字符串创建右部为单个符号的产生式
    Production(std::string left, std::string right) {
        this->left = std::move(left);
        this->right.push_back(std::move(right));
    }
    // 构造函数：从一个符号向量创建产生式
    Production(std::string left, std::vector<std::string> right) : left(std::move(left)), right(std::move(right)) {}
    // 重载比较运算符，用于判断两个产生式是否相等
    bool operator==(const Production& other) const {
        return left == other.left && right == other.right;
    }
};

    std::vector<std::string> non_terminal_sym;   // 非终结符集合
    std::vector<Production> production;          // 产生式集合
    std::string start_sym;                       // 起始符号
};

    std::vector<Production> project_set;         // 该状态包含的LR(0)项目集
    // 构造函数
    DFAState(std::string name, const std::vector<Production>& projects) : state_name(std::move(name)), project_set(projects) {}
};

// DFA状态转移函数的结构体
struct StateTransFunc {
    DFAState current_state;                      // 当前状态
    std::string input_sym;                       // 输入符号
    std::vector<DFAState> next_state;            // 下一状态 (理论上是集合，但此处通常只有一个)
    StateTransFunc() = delete;                   // 删除默认构造函数，强制使用带参构造
    // 带参数的构造函数
    StateTransFunc(DFAState current_state, std::string input_sym, std::vector<DFAState> next_state)
        : current_state(std::move(current_state)), input_sym(std::move(input_sym)), next_state(std::move(next_state)) {}
};

    std::vector<DFAState> start_state;           // 初态集合 (通常只有一个)
    std::vector<DFAState> final_state;           // 终态集合
};

// 实现：从文件读取文法数据
void input_data(Grammar& grammar) {
    try {
        std::ifstream input_file("../input.txt"); // 尝试打开输入文件
        std::string line; // 存储读取的每一行
        std::string temp; // 临时变量，用于存储分割出的单词

        // 读取非终结符数量 (并丢弃)
        std::getline(input_file, line);
        // 读取非终结符列表
        std::getline(input_file, line);
        std::istringstream iss(line); // 创建字符串流以解析该行
        while(iss >> temp) { // 循环读取每个单词
            grammar.non_terminal_sym.push_back(temp); // 添加到非终结符向量
        }
        std::getline(input_file, line); // 跳过空行

        // 读取终结符数量 (并丢弃)
        std::getline(input_file, line);
        // 读取终结符列表
        std::getline(input_file, line);
        iss = std::istringstream(line); // 重新初始化字符串流
        while(iss >> temp) {
            grammar.terminal_sym.push_back(temp); // 添加到终结符向量
        }
        std::getline(input_file, line); // 跳过空行

        // 读取产生式数量
        std::getline(input_file, line);
        int num_production = std::stoi(line); // 将字符串转换为整数
        for (int i = 0; i < num_production; i++) {
            std::getline(input_file, line); // 读取一条产生式
            iss = std::istringstream(line); // 初始化字符串流
            std::vector<std::string> syms; // 存储该行分割出的所有符号
            while (iss >> temp) {
                syms.push_back(temp);
            }
            // 添加错误检查：确保行至少有3个部分且第二个是 "->"
            if (syms.size() >= 3 && syms[1] == "->") {
                Production new_p;
                new_p.left = syms[0]; // 左部是非终结符
                // 右部从第三个符号开始到最后
                for (int count = 2; count < syms.size(); count++) {
                    new_p.right.push_back(syms[count]);
                }
                grammar.production.push_back(new_p); // 添加到产生式集合
            } else {
                std::cerr << "产生式格式错误！原始行数据如下：" << line << std::endl;
            }
        }
        std::getline(input_file, line); // 跳过空行

        // 读取起始符号
        std::getline(input_file, line);
        grammar.start_sym = line;

    } catch (const std::exception& e) { // 如果文件读取失败 (如文件不存在)
        std::cerr << "文件读取失败：" << e.what() << std::endl;
        std::cout << "使用默认文法进行测试：" << std::endl;
        // 使用一个简单的S -> A, A -> aA | b 文法作为测试
        grammar.non_terminal_sym = {"S", "A"};
        grammar.terminal_sym = {"a", "b", "#"}; // "#" 代表输入结束符
        grammar.start_sym = "S";
        Production p1{"S", {"A"}};
        Production p2{"A", {"a", "A"}};
        Production p3{"A", {"b"}};
        grammar.production = {p1, p2, p3};
    }
}

// 实现：生成所有LR(0)项目
void generate_project(std::unordered_map<std::string, std::vector<Production>>& project_map, Grammar& grammar){
    // 检查是否存在拓广产生式 S' -> ...
    bool flag = false;
    for (const auto& production : grammar.production) {
        if (production.left == "S'") {
            flag = true;
            break;
        }
    }
    // 如果没有，则添加拓广产生式 S' -> start_symbol
    if (!flag) {
        Production start_production;
        start_production.left = "S'";
        start_production.right.push_back(grammar.start_sym);
        // 将拓广产生式插入到产生式列表的最前面
        grammar.production.insert(grammar.production.begin(), start_production);
    }

    // 为每条产生式生成所有可能的项目 (点号在不同位置)
    for (const auto& production : grammar.production) {
        // 对于长度为n的右部，有 n+1 个项目 (点号可以放在n+1个位置)
        for (size_t i = 0; i <= production.right.size(); ++i) {
            Production new_production;
            new_production.left = production.left; // 新项目的左部与原产生式相同
            new_production.right.clear(); // 清空右部

            // 在位置i处插入点号 "."
            for (size_t j = 0; j < production.right.size(); ++j) {
                if (j == i) {
                    new_production.right.emplace_back("."); // 先插入点号
                }
                new_production.right.push_back(production.right[j]); // 再插入原符号
            }
            // 如果点号在末尾 (i == production.right.size())，则最后再添加一个点号
            if (i == production.right.size()) {
                new_production.right.emplace_back(".");
            }
            // 将生成的项目存入project_map，以左部为键
            project_map[new_production.left].push_back(new_production);
        }
    }
}

// 实现：生成项目集规范族和DFA
void generate_project_set(DFA& dfa, const std::unordered_map<std::string, std::vector<Production>>& project_map, const Grammar& grammar){
    // 存储所有已发现的项目集 (每个项目集是一个Production的vector)
    std::vector<std::vector<Production>> project_sets;
    // 标记每个项目集是否已被处理过
    std::vector<bool> processed_sets;

    // 创建初始项目集 I0 = Closure({S' -> .S})
    std::vector<Production> initial_project;
    initial_project.push_back(project_map.at("S'")[0]); // 获取S'的第一个项目 (S' -> .start_symbol)
    std::vector<Production> closure_initial = calc_closure(initial_project, project_map, grammar); // 计算其闭包
    project_sets.push_back(closure_initial); // 将初始项目集加入集合
    processed_sets.push_back(false); // 标记为未处理

    // 收集所有符号 (终结符 + 非终结符)
    std::set<std::string> all_symbols;
    for (const auto& term : grammar.terminal_sym) {
        all_symbols.insert(term);
    }
    for (const auto& non_term : grammar.non_terminal_sym) {
        all_symbols.insert(non_term);
    }

    // 用于记录状态转移的映射：from_state_index -> symbol -> to_state_index
    std::unordered_map<size_t, std::unordered_map<std::string, size_t>> transitions_map;

    // 使用队列来实现BFS，逐个处理项目集
    std::queue<size_t> work_queue;
    work_queue.push(0); // 将初始项目集 (索引0) 加入待处理队列

    while (!work_queue.empty()) {
        size_t current_index = work_queue.front();
        work_queue.pop();

        if (processed_sets[current_index]) { // 如果已处理，跳过
            continue;
        }
        processed_sets[current_index] = true; // 标记为已处理

        // 对当前项目集，尝试用每个符号X进行GO操作
        for (const auto& symbol : all_symbols) {
            std::vector<Production> go_result = GO(project_sets[current_index], symbol, project_map, grammar);
            if (go_result.empty()) { // 如果结果为空，说明没有这个转移
                continue;
            }

            // 检查GO结果是否是新的项目集
            bool is_new = true;
            size_t existing_index = 0;
            for (size_t i = 0; i < project_sets.size(); ++i) {
                if (is_project_set_equal(go_result, project_sets[i])) {
                    is_new = false; // 找到了相同的项目集
                    existing_index = i;
                    break;
                }
            }

            if (is_new) {
                project_sets.push_back(go_result); // 添加新项目集
                processed_sets.push_back(false); // 标记为未处理
                work_queue.push(project_sets.size() - 1); // 将新项目集加入待处理队列
                existing_index = project_sets.size() - 1; // 新项目集的索引
            }

            // 记录从current_index状态，经symbol，转移到existing_index状态
            transitions_map[current_index][symbol] = existing_index;
        }
    }

    // --- 构建DFA对象 ---
    // 添加所有状态
    for (size_t i = 0; i < project_sets.size(); ++i) {
        dfa.states.emplace_back(std::to_string(i), project_sets[i]); // 状态名为 "0", "1", ...
    }
    // 添加字母表
    dfa.alphabet.insert(dfa.alphabet.end(), all_symbols.begin(), all_symbols.end());
    // 设置初态 (第一个状态)
    if (!dfa.states.empty()) {
        dfa.start_state.push_back(dfa.states[0]);
    }
    // 设置终态：任何包含规约项目 (点号在末尾) 的状态都是终态
    for (size_t i = 0; i < dfa.states.size(); ++i) {
        bool is_final = false;
        for (const auto& project : dfa.states[i].project_set) {
            if (!project.right.empty() && project.right.back() == ".") {
                is_final = true;
                break;
            }
        }
        if (is_final) {
            dfa.final_state.push_back(dfa.states[i]);
        }
    }
    // 添加状态转移关系
    for (const auto& transition_entry : transitions_map) {
        size_t from_state = transition_entry.first;
        for (const auto& symbol_transition : transition_entry.second) {
            std::string symbol = symbol_transition.first;
            size_t to_state = symbol_transition.second;
            if (from_state < dfa.states.size() && to_state < dfa.states.size()) {
                // 创建一个转移函数对象并添加到DFA
                StateTransFunc trans_func(dfa.states[from_state], symbol, {dfa.states[to_state]});
                dfa.transitions.push_back(trans_func);
            }
        }
    }
}

// 实现：计算GO(I, X)函数
std::vector<Production> GO(const std::vector<Production>& project_set, const std::string& symbol,
                            const std::unordered_map<std::string, std::vector<Production>>& project_map, const Grammar& grammar) {
    std::vector<Production> result;

    // 遍历项目集I中的每个项目
    for (const auto& project : project_set) {
        // 查找点号 "." 后面紧跟符号X的项目
        for (size_t i = 0; i < project.right.size(); ++i) {
            if (project.right[i] == "." && i + 1 < project.right.size() && project.right[i + 1] == symbol) {
                Production moved_project = project; // 复制原项目
                // 将点号右移一位：也就是将X的位置与点号对调
                moved_project.right[i] = symbol;
                moved_project.right[i + 1] = ".";
                result.push_back(moved_project); // 将移动后的项目加入结果
            }
        }
    }

    // 对移动后得到的项目集计算闭包
    return calc_closure(result, project_map, grammar);
}

// 实现：计算一个项目集的闭包 (Closure)
std::vector<Production> calc_closure(const std::vector<Production>& project_set,
                                    const std::unordered_map<std::string, std::vector<Production>>& project_map,
                                    const Grammar& grammar){
    std::vector<Production> result = project_set; // 初始闭包就是项目集本身
    bool changed = true; // 标志位，表示本次迭代是否有新项目加入

    while (changed) {
        changed = false;
        std::vector<Production> new_projects; // 存储本轮迭代发现的新项目

        // 遍历当前闭包中的每个项目
        for (const auto& project : result) {
            // 查找点号 "." 后面紧跟一个非终结符A的情况
            for (size_t i = 0; i < project.right.size(); ++i) {
                if (project.right[i] == "." && i + 1 < project.right.size()) {
                    std::string next_symbol = project.right[i + 1];
                    // 检查next_symbol是否为非终结符
                    if (std::find(grammar.non_terminal_sym.begin(), grammar.non_terminal_sym.end(), next_symbol) != grammar.non_terminal_sym.end()) {
                        // 获取该非终结符A的所有产生式对应的项目 (点号在最前面)
                        auto it = project_map.find(next_symbol);
                        if (it != project_map.end()) {
                            for (const auto& prod : it->second) {
                                // 只取点号在最前面的项目 (即A -> .α 形式)
                                if (!prod.right.empty() && prod.right[0] == ".") {
                                    // 检查该项目是否已在result中，避免重复
                                    bool exists = false;
                                    for (const auto& existing : result) {
                                        if (existing.left == prod.left && existing.right == prod.right) {
                                            exists = true;
                                            break;
                                        }
                                    }
                                    if (!exists) {
                                        new_projects.push_back(prod);
                                        changed = true; // 发现了新项目
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
        // 将本轮发现的所有新项目一次性加入result
        result.insert(result.end(), new_projects.begin(), new_projects.end());
    }
    return result;
}

// 实现：比较两个项目集是否相等
bool is_project_set_equal(const std::vector<Production>& set1, const std::vector<Production>& set2) {
    if (set1.size() != set2.size()) {
        return false;
    }

    // 为了忽略顺序的影响，先对两个项目集进行排序
    std::vector<Production> sorted_set1 = set1;
    std::vector<Production> sorted_set2 = set2;

    // 定义排序规则：先按左部排序，再按右部长度排序，最后按右部内容排序
    auto prod_cmp = [](const Production& a, const Production& b) {
        if (a.left != b.left) return a.left < b.left;
        if (a.right.size() != b.right.size()) return a.right.size() < b.right.size();
        return a.right < b.right;
    };

    std::sort(sorted_set1.begin(), sorted_set1.end(), prod_cmp);
    std::sort(sorted_set2.begin(), sorted_set2.end(), prod_cmp);

    // 逐个比较排序后的项目
    for (size_t i = 0; i < sorted_set1.size(); ++i) {
        if (sorted_set1[i].left != sorted_set2[i].left || sorted_set1[i].right != sorted_set2[i].right) {
            return false;
        }
    }
    return true;
}

// 实现：生成LR(0)分析表
void generate_lr0_table(const DFA& dfa, const Grammar& grammar, LR0Table& lr0_table) {
    // --- 初始化分析表 ---
    for (const auto& dfa_state : dfa.states) {
        std::string state = dfa_state.state_name;
        // 初始化Action表：对所有终结符和#设置为ERR
        for (const auto& terminal : grammar.terminal_sym) {
            lr0_table.action_table[state][terminal] = TableValue(ERR, -1);
        }
        lr0_table.action_table[state]["#"] = TableValue(ERR, -1);
        // 初始化Goto表：对所有非终结符设置为"-1" (表示错误)
        for (const auto& non_terminal : grammar.non_terminal_sym) {
            lr0_table.goto_table[state][non_terminal] = "-1";
        }
    }

    // --- 填充移进(SHIFT)和转移(GOTO)项 ---
    // 遍历DFA的所有转移函数
    for (const auto& transition : dfa.transitions) {
        std::string from_state = transition.current_state.state_name;
        std::string symbol = transition.input_sym;
        std::string to_state = transition.next_state[0].state_name;

        if (std::find(grammar.terminal_sym.begin(), grammar.terminal_sym.end(), symbol) != grammar.terminal_sym.end()) {
            // 如果符号是终结符，则填入Action表，动作为SHIFT，下一状态为to_state
            lr0_table.action_table[from_state][symbol] = TableValue(SHIFT, std::stoi(to_state));
        } else if (std::find(grammar.non_terminal_sym.begin(), grammar.non_terminal_sym.end(), symbol) != grammar.non_terminal_sym.end()) {
            // 如果符号是非终结符，则填入Goto表，下一状态为to_state
            lr0_table.goto_table[from_state][symbol] = to_state;
        }
    }

    // --- 填充规约(REDUCE)和接受(ACCEPT)项 ---
    // 遍历DFA的每个状态
    for (size_t state_idx = 0; state_idx < dfa.states.size(); ++state_idx) {
        std::string state = dfa.states[state_idx].state_name;
        const std::vector<Production>& project_set = dfa.states[state_idx].project_set;

        // 遍历该状态的每个项目
        for (const auto& project : project_set) {
            // 查找点号 "." 的位置
            size_t dot_pos = std::find(project.right.begin(), project.right.end(), ".") - project.right.begin();
            if (dot_pos >= project.right.size()) continue; // 如果没找到点号，跳过

            // 如果点号在最右边 (即dot_pos + 1 >= size)，这是一个规约或接受项目
            if (dot_pos + 1 >= project.right.size()) {
                // 情况1：接受项目 (Accept) - S' -> α.
                // 检查是否为拓广产生式 S' -> ... .
                if (project.left == "S'" && project.right.size() == 2 && project.right[1] == ".") {
                    lr0_table.action_table[state]["#"] = TableValue(ACCEPT, -1);
                } else {
                    // 情况2：规约项目 (Reduce) - A -> α.
                    // 构造去除点号后的原始右部
                    std::vector<std::string> original_right;
                    for (const auto& sym : project.right) {
                        if (sym != ".") {
                            original_right.push_back(sym);
                        }
                    }
                    // 在原始文法中查找该产生式对应的索引
                    int production_index = -1;
                    for (size_t i = 0; i < grammar.production.size(); ++i) {
                        if (grammar.production[i].left == project.left && grammar.production[i].right == original_right) {
                            production_index = i;
                            break;
                        }
                    }
                    if (production_index != -1) {
                        // 对所有终结符 (包括 #) 填入规约动作
                        // 注意：如果该位置已有移进动作 (来自GOTO填充)，这里不会覆盖，从而检测冲突
                        for (const auto& terminal : grammar.terminal_sym) {
                            if (lr0_table.action_table[state][terminal].movement == ERR) {
                                lr0_table.action_table[state][terminal] = TableValue(REDUCE, production_index);
                            }
                        }
                        if (lr0_table.action_table[state]["#"].movement == ERR) {
                            lr0_table.action_table[state]["#"] = TableValue(REDUCE, production_index);
                        }
                    }
                }
            }
        }
    }
}

// 实现：检查文法是否为LR(0)文法 (即分析表无冲突)
bool is_lr0_grammar(const DFA& dfa, const Grammar& grammar) {
    // 遍历DFA的每个状态
    for (const auto& dfa_state : dfa.states) {
        const std::vector<Production>& project_set = dfa_state.project_set;
        std::vector<Production> shift_projects; // 存储移进项目 (点号不在末尾)
        std::vector<Production> reduce_projects; // 存储规约项目 (点号在末尾)

        for (const auto& project : project_set) {
            size_t dot_pos = std::find(project.right.begin(), project.right.end(), ".") - project.right.begin();
            if (dot_pos >= project.right.size()) continue;

            if (dot_pos + 1 < project.right.size()) {
                // 点号后面还有符号，是移进项目
                shift_projects.push_back(project);
            } else {
                // 点号在末尾，是规约项目
                reduce_projects.push_back(project);
            }
        }

        // 冲突检查1：移进-规约冲突
        // 如果一个状态同时包含移进项目和规约项目，则存在移进-规约冲突
        if (!shift_projects.empty() && !reduce_projects.empty()) {
            return false;
        }

        // 冲突检查2：规约-规约冲突
        // 如果一个状态包含多个不同的规约项目，则存在规约-规约冲突
        if (reduce_projects.size() > 1) {
            // 检查所有规约项目是否都对应同一个原始产生式
            for (size_t i = 0; i < reduce_projects.size(); i++) {
                for (size_t j = i + 1; j < reduce_projects.size(); j++) {
                    // 提取两个规约项目的原始右部 (去掉点号)
                    std::vector<std::string> right1, right2;
                    for (const auto& sym : reduce_projects[i].right) if (sym != ".") right1.push_back(sym);
                    for (const auto& sym : reduce_projects[j].right) if (sym != ".") right2.push_back(sym);
                    // 如果左部不同 或 右部不同，则是不同的产生式，存在冲突
                    if (reduce_projects[i].left != reduce_projects[j].left || right1 != right2) {
                        return false;
                    }
                }
            }
        }
    }
    // 如果所有状态都通过检查，则是LR(0)文法
    return true;
}

// 实现：打印LR(0)分析表
void print_lr0_table(const LR0Table& lr0_table, const DFA& dfa, const Grammar& grammar) {
    // 打印ACTION表
    std::cout << "\nACTION 表:\n";
    std::cout << "State\t"; // 表头第一列
    for (const auto& terminal : grammar.terminal_sym) {
        std::cout << terminal << "\t"; // 打印每个终结符
    }
    std::cout << "#\t"; // 打印输入结束符#
    std::cout << "\n"; // 换行

    // 打印每一行 (每个状态)
    for (const auto& dfa_state : dfa.states) {
        std::string state = dfa_state.state_name;
        std::cout << state << "\t"; // 打印状态名

        // 打印该状态对每个终结符的动作
        for (const auto& terminal : grammar.terminal_sym) {
            TableValue val = lr0_table.action_table.at(state).at(terminal);
            switch (val.movement) {
                case SHIFT:
                    std::cout << "s" << val.nextState << "\t"; // sN
                    break;
                case REDUCE:
                    std::cout << "r" << val.nextState << "\t"; // rN (N为产生式索引)
                    break;
                case ACCEPT:
                    std::cout << "acc\t"; // acc
                    break;
                case ERR:
                    std::cout << "err\t"; // err
                    break;
            }
        }
        // 处理#符号
        TableValue val = lr0_table.action_table.at(state).at("#");
        switch (val.movement) {
            case SHIFT:
                std::cout << "s" << val.nextState << "\t";
                break;
            case REDUCE:
                std::cout << "r" << val.nextState << "\t";
                break;
            case ACCEPT:
                std::cout << "acc\t";
                break;
            case ERR:
                std::cout << "err\t";
                break;
        }
        std::cout << "\n"; // 换行，进入下一行
    }

    // 打印GOTO表
    std::cout << "\nGOTO 表:\n";
    std::cout << "State\t";
    for (const auto& non_terminal : grammar.non_terminal_sym) {
        std::cout << non_terminal << "\t"; // 打印每个非终结符
    }
    std::cout << "\n";

    for (const auto& dfa_state : dfa.states) {
        std::string state = dfa_state.state_name;
        std::cout << state << "\t";

        // 打印该状态对每个非终结符的转移
        for (const auto& non_terminal : grammar.non_terminal_sym) {
            std::string val = lr0_table.goto_table.at(state).at(non_terminal);
            if (val != "-1") {
                std::cout << val << "\t"; // 输出目标状态
            } else {
                std::cout << "err\t"; // err
            }
        }
        std::cout << "\n";
    }
}

// 实现：打印DFA的所有项目集
void print_project_set(const DFA& dfa) {
    std::cout << "\n[LR(0) item set cluster]\n";
    for (int i = 0; i < dfa.states.size(); i++) {
        std::cout << i << ": "; // 打印状态索引
        for (const auto& project : dfa.states[i].project_set) {
            std::cout << project.left << " -> ";
            for (const auto& sym : project.right) {
                std::cout << sym << " "; // 打印产生式右部
            }
            // 如果这不是最后一个项目，打印分隔符 "| "
            if (is_production_equals(project, dfa.states[i].project_set.back()) == false) {
                std::cout << "| ";
            }
        }
        std::cout << std::endl;
    }
}

// 实现：打印DFA的状态转移函数
void print_dfa_trans_func(const DFA& dfa) {
    std::cout << "\n[LR(0) state tran function]\n";
    for (const auto& t : dfa.transitions) {
        std::cout << t.current_state.state_name << ", " << t.input_sym << "\t-> ";
        for (const auto& n : t.next_state) {
            std::cout << n.state_name << " ";
            // 如果这不是最后一个状态，打印逗号
            if (&n != &t.next_state.back()) {
                std::cout << ", ";
            }
        }
        std::cout << std::endl;
    }
}

// 实现：判断两个产生式是否相等
bool is_production_equals(const Production& p1, const Production& p2) {
    if (p1.left != p2.left) {
        return false;
    }
    if (p1.right.size() != p2.right.size()) {
        return false;
    }
    for (size_t i = 0; i < p1.right.size(); i++) {
        if (p1.right[i] != p2.right[i]) {
            return false;
        }
    }
    return true;
}
